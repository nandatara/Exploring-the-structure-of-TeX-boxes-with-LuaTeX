\dobox{\vbox to 25pt{A}}

\dobox{\vtop to 25pt{A\hfill B}}
\dobox{\hbox to 25pt{A\hskip 10pt\hfill B}}

D\hskip 0pt plus 3fill}}
\dobox{\hbox{$\displaystyle \int f(x) dx$}}
